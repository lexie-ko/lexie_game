                else {

                    if (this.state.selectedPlayers.length >= 16) return;

                    this.state.selectedPlayers.push({ poolIdx: idx, data: this.state.activePool[idx], pos: 0, finished: false, stun: 0, stunCount: 0 });

                }

                this.dom.count.innerText = this.state.selectedPlayers.length;

                this.renderGrid();

            },


            resetSelection: function() {

                this.state.selectedPlayers = [];

                this.dom.count.innerText = "0";

                this.renderGrid();

            },


            startGame: function() {

                if (this.state.selectedPlayers.length < 2) { alert("2명 이상의 캐릭터를 선택해주세요!"); return; }

                const isBalanced = document.getElementById('balance-check').checked;

                this.state.luckyIdx = Math.floor(Math.random() * this.state.selectedPlayers.length);


                this.state.selectedPlayers.forEach((p, i) => {

                    p.currentSpeed = isBalanced ? 1.0 : p.data.speed;

                    p.pos = 0; p.finished = false; p.stun = 0; p.stunCount = 0;

                    p.isLucky = (i === this.state.luckyIdx);

                });

                

                this.dom.btnShowResults.classList.add('hidden');

                this.dom.btnRestartTop.classList.add('hidden');

                this.dom.raceStatus.innerText = "RACING...";

                this.dom.screens.select.classList.add('hidden');

                this.dom.screens.race.classList.remove('hidden');

                this.renderTrack();

                this.state.isRacing = true;

                this.state.winners = [];

                this.state.speedMultiplier = 1;

                requestAnimationFrame(() => this.gameLoop());

            },


            renderTrack: function() {

                this.dom.track.innerHTML = '';

                this.state.selectedPlayers.forEach((p, i) => {

                    const lane = document.createElement('div');

                    lane.className = 'lane flex-shrink-0 h-11 sm:h-12 w-full relative border-b border-white/5';

                    lane.innerHTML = `

                        <div class="absolute left-1 top-1/2 -translate-y-1/2 text-white/10 font-black text-xs z-0">${i + 1}</div>

                        <div id="runner-${i}" class="runner-avatar absolute left-0 top-1/2 -translate-y-1/2 z-10 flex flex-col items-center justify-center w-10" style="transform: translate(0px, -50%)">

                            <div class="relative flex flex-col items-center char-body-container">

                                <div class="w-8 h-8 flex items-center justify-center relative">

                                    ${p.data.imgUrl ? `<img src="${p.data.imgUrl}" class="max-w-full max-h-full object-contain">` : `<span class="text-xl sm:text-2xl">${p.data.icon}</span>`}

                                    <div class="ice-block"></div>

                                </div>

                                <div class="relative w-full h-1">

                                    <div class="leg leg-left"></div>

                                    <div class="leg leg-right"></div>

                                    <div class="surfboard"></div>

                                </div>

                                <div id="ufo-${i}" class="ufo-beam absolute -top-8 left-1/2 -translate-x-1/2 w-10 h-16 hidden"></div>

                            </div>

                            <div class="text-[6px] bg-black/60 px-0.5 rounded text-white whitespace-nowrap mt-0.5 tracking-tighter opacity-80">${p.data.name}</div>

                        </div>

                        <div id="rank-display-${i}" class="absolute right-7 top-1/2 -translate-y-1/2 font-black text-2xl text-yellow-400 hidden z-30 drop-shadow-md italic"></div>

                        <div class="absolute right-0.5 top-1/2 -translate-y-1/2 text-lg opacity-80 z-20">🏁</div>

                    `;

                    this.dom.track.appendChild(lane);

                });

            },


            backToSelect: function() {

                this.state.isRacing = false;

                this.dom.screens.race.classList.add('hidden');

                this.dom.screens.select.classList.remove('hidden');

            },


            togglePause: function() {

                this.state.isPaused = !this.state.isPaused;

                document.getElementById('btn-pause').innerText = this.state.isPaused ? "▶" : "⏸";

            },


            toggleSpeed: function() {

                this.state.speedMultiplier = this.state.speedMultiplier === 1 ? 2 : (this.state.speedMultiplier === 2 ? 4 : 1);

                document.getElementById('btn-speed').innerText = `${this.state.speedMultiplier}x`;

            },


            gameLoop: function() {

                if (!this.state.isRacing) return;

                if (!this.state.isPaused) {

                    let allFinished = true;

                    const trackWidth = this.dom.track.clientWidth - 40;


                    this.state.selectedPlayers.forEach((p, i) => {

                        if (p.finished) return;

                        allFinished = false;


                        const runnerEl = document.getElementById(`runner-${i}`);

                        const ufoEl = document.getElementById(`ufo-${i}`);

                        const rankEl = document.getElementById(`rank-display-${i}`);

                        const container = runnerEl.querySelector('.char-body-container');


                        if (p.stun > 0) {

                            p.stun--;

                            if (p.stun <= 0) {

                                ufoEl.classList.add('hidden');

                                container.classList.remove('is-frozen', 'is-blackhole', 'is-wave');

                            }

                            return;

                        }


                        if (!p.isLucky && p.stunCount < 2) {

                            let triggerChance = 0.003 * this.state.speedMultiplier;

                            const progress = p.pos / trackWidth;

                            if (p.stunCount === 0 && progress > 0.5 && progress < 0.85) triggerChance = 0.015;


                            if (Math.random() < triggerChance) {

                                let handicap;

                                if (Math.random() < 0.45) {

                                    if (this.state.currentMap === 'sea') {

                                        handicap = { id: 'wave', icon: '🌊', text: '파도 밀림!', duration: 60, class: 'is-wave', back: 35 };

                                    } else {

                                        handicap = { id: 'blackhole', icon: '🕳️', text: '블랙홀!', duration: 60, class: 'is-blackhole' };

                                    }

                                } else {

                                    handicap = SHARED_HANDICAPS[Math.floor(Math.random() * SHARED_HANDICAPS.length)];

                                }


                                p.stun = handicap.duration / this.state.speedMultiplier;

                                p.stunCount++;

                                this.showToast(`${p.data.icon} ${handicap.text}`);

                                if (handicap.beam) ufoEl.classList.remove('hidden');

                                if (handicap.class) container.classList.add(handicap.class);

                                if (handicap.id === 'ufo') p.pos = Math.max(0, p.pos - 35);

                                if (handicap.back) p.pos = Math.max(0, p.pos - handicap.back);

                            }

                        }


                        let moveStep = (p.currentSpeed * 0.9) + (Math.random() * 0.4); 

                        moveStep *= this.state.speedMultiplier;

                        p.pos += moveStep;


                        if (p.pos >= trackWidth) {

                            p.pos = trackWidth;

                            p.finished = true;

                            this.state.winners.push(p);

                            rankEl.innerText = this.state.winners.length;

                            rankEl.classList.remove('hidden');

                            runnerEl.style.opacity = "0.5";

                        }

                        runnerEl.style.transform = `translate(${p.pos}px, -50%)`;

                    });


                    if (allFinished) { this.onRaceFinish(); return; }

                }

                requestAnimationFrame(() => this.gameLoop());

            },


            showToast: function(msg) {

                const t = document.getElementById('event-toast');

                t.innerHTML = msg; t.classList.remove('hidden');

                setTimeout(() => t.classList.add('hidden'), 1500);

            },


            onRaceFinish: function() {

                this.state.isRacing = false;

                this.dom.raceStatus.innerText = "FINISH!";

                this.dom.btnShowResults.classList.remove('hidden');

                this.dom.btnRestartTop.classList.remove('hidden');

            },


            showResults: function() {

                const list = this.dom.resultList; list.innerHTML = '';

                this.state.winners.forEach((p, i) => {

                    const isLast = i === this.state.winners.length - 1;

                    const row = document.createElement('div');

                    let bg = i === 0 ? "bg-yellow-400 text-slate-900 border-2 border-white/50" : (isLast ? "bg-red-600 animate-pulse text-white" : "bg-slate-700/80 text-white shadow-sm");

                    row.className = `flex items-center p-2 rounded-xl ${bg} shadow-lg transition-all duration-300`;

                    

                    const charThumb = p.data.imgUrl ? `<img src="${p.data.imgUrl}" class="w-8 h-8 object-contain mx-2">` : `<div class="text-2xl mx-2">${p.data.icon}</div>`;

                    

                    row.innerHTML = `

                        <div class="font-black w-8 text-center text-sm">${isLast ? '💥' : (i+1)}</div>

                        ${charThumb}

                        <div class="flex-1 font-bold text-[11px] truncate tracking-tight">${p.data.name}</div>

                        <div class="text-[9px] font-black mr-1 uppercase">${isLast ? '꽝' : ''}</div>

                    `;

                    list.appendChild(row);

                });

                this.dom.modal.classList.remove('hidden');

            },


            closeResults: function() {

                this.dom.modal.classList.add('hidden');

            },


            resetToLobby: function() {

                this.dom.modal.classList.add('hidden');

                this.dom.screens.race.classList.add('hidden');

                this.dom.screens.select.classList.remove('hidden');

                this.dom.btnShowResults.classList.add('hidden');

                this.dom.btnRestartTop.classList.add('hidden');

                this.shufflePool();

                this.resetSelection();

            }

        };


        window.onload = () => app.init();

    </script>
